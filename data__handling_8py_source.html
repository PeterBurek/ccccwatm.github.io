329 def mapattrNetCDF(name):
330  """
331  get the map attributes like col, row etc from a ntcdf map
332  and define the rectangular of the mask map inside the netcdf map
333  """
334  filename = os.path.splitext(name)[0] + '.nc'

338  msg = "Checking netcdf map \n"
339  raise CWATMFileError(filename,msg)
340 
341  x1 = round(nf1.variables.values()[0][0],5)
342  x2 = round(nf1.variables.values()[0][1],5)
343  xlast = round(nf1.variables.values()[0][-1],5)
344  y1 = round(nf1.variables.values()[1][0],5)
345  ylast = round(nf1.variables.values()[1][-1],5)
346  cellSize = round(np.abs(x2 - x1),5)
347  nrRows = int(0.5+np.abs(ylast - y1) / cellSize + 1)
348  nrCols = int(0.5+np.abs(xlast - x1) / cellSize + 1)
349  x = x1 - cellSize / 2
350  y = y1 + cellSize / 2
351  nf1.close()
352 
353  cut0 = int(np.abs(maskmapAttr['x'] - x) / maskmapAttr['cell'])
354  cut1 = cut0 + maskmapAttr['col']
355  cut2 = int(np.abs(maskmapAttr['y'] - y) / maskmapAttr['cell'])
356  cut3 = cut2 + maskmapAttr['row']

406

409  except:
410  msg = "Netcdf map stacks: \n"
411  raise CWATMFileError(filename,msg)
412 
413 
414  # date if used daily, monthly or yearly or day of year
415  if useDaily == "DOY": # day of year 1-366

417  else:
